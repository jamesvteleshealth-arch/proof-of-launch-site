Fix the ContactForm with a React 18-compatible pattern. Two good options:
Option A (recommended — simplest, most reliable): Use useState + useTransition + direct server action invocation:
tsx"use client";
import { useState, useTransition } from "react";
import { submitContact, type ContactResult } from "@/app/actions";
import { cn } from "@/lib/cn";

export function ContactForm() {
  const [result, setResult] = useState<ContactResult | null>(null);
  const [isPending, startTransition] = useTransition();

  async function handleSubmit(formData: FormData) {
    startTransition(async () => {
      const res = await submitContact(formData);
      setResult(res);
    });
  }

  return (
    <form action={handleSubmit}>
      {/* ... form fields ... */}
      <button
        type="submit"
        disabled={isPending}
        className={cn(
          "...",
          result?.success && "bg-teal-dim"
        )}
      >
        {result?.success ? "✓ Received — We'll be in touch within 24 hours" : "Send — We'll Respond Within 24 Hours →"}
      </button>
    </form>
  );
}
This uses the action={fn} prop on <form> (a stable React 18 + Next.js 14 feature) and tracks pending/success state with useTransition + useState.
Option B (if you need progressive enhancement): Use a traditional onSubmit handler, manually call the server action, manage state with useState. Slightly more boilerplate but works on every React version.
